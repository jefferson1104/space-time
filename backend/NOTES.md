### COMMANDS
```bash
# start project
$ npm init -y

# install typescript
$ npm i typescript -D

# run typescript to create tsconfig.json
$ npx tsc --init

# install tsx library
$ npm i tsx -D

# install fastify
$ npm i fastify

# install ESLint
$ npm i eslint -D

# install Rocketseat ESLint config
$ npm i @rocketseat/eslint-config -D

# install prisma
$ npm i prisma -D

# start prisma in the project using SQLite
$ npx prisma init --datasource-provider SQLite

# prisma create database migration
$ npx prisma migrate dev

# open prisma studio GUI
$ npx prisma studio

# install prisma client
$ npm i @prisma/client
```
